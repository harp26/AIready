One-time Google Sheet Setup

Connect your Google Sheet in 5 steps

1
Create a blank Google Sheet in your Google account. Name it anything — e.g. "AI Survey Responses".
2
In the Sheet, click Extensions → Apps Script. A new tab opens with a code editor.
3
Delete all existing code. Paste the script below exactly as-is. Then click Save (💾).
const SHEET_NAME = "Responses"; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ "Timestamp","Batch","Name","Stream", "Study Medium","Background","AI Layer", "Tools Used","Usage Purpose","Language Used", "Interaction Mode","Detailed Prompting", "Power Uses","Mentor Willing","Open Answer" ]); sheet.setFrozenRows(1); } const data = JSON.parse(e.postData.contents); sheet.appendRow([ new Date().toLocaleString("en-IN"), data.batch || "Not set", data.name || "Anonymous", data.stream || "", data.medium || "", data.background || "", data.layer || "", (data.tools || []).join(", "), (data.purpose || []).join(", "), data.langUsed || "", data.interact || "", data.detailedPrompt || "", (data.powerUse || []).join(", "), data.mentor || "", data.openAnswer || "" ]); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet() { return ContentService.createTextOutput("Survey endpoint is active."); }

4
Click Deploy → New deployment. Set Type to Web app. Set "Who has access" to Anyone. Click Deploy. Authorize when prompted.
5
Copy the Web App URL that appears. Paste it in the field above and click Save.
